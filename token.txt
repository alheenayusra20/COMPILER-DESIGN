/* Filename: tokenizer.l */

%{
#include <stdio.h>
#include <string.h>
%}

%option noyywrap

%%
"int"|"float"|"char"|"if"|"else"|"while"|"for"|"return"   { printf("Keyword: %s\n", yytext); }
[0-9]+                                                    { printf("Number: %s\n", yytext); }
[+\-*/=<>]                                                { printf("Operator: %s\n", yytext); }
[a-zA-Z_][a-zA-Z0-9_]*                                    { printf("Identifier: %s\n", yytext); }
";"                                                       { printf("Semicolon: %s\n", yytext); }
"("                                                       { printf("Left Parenthesis: %s\n", yytext); }
")"                                                       { printf("Right Parenthesis: %s\n", yytext); }
"{"                                                       { printf("Left Brace: %s\n", yytext); }
"}"                                                       { printf("Right Brace: %s\n", yytext); }
[ \t\n]+                                                  { /* Ignore whitespace */ }
.                                                         { printf("Unknown: %s\n", yytext); }
%%

int main() {
    yylex();
    return 0;
}
